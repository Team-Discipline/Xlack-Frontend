package hw.account;

public class Bank {
	public static void main(String args[]) {
		Account[] array = new Account[5];
		array[0] = new Account("1", "user", 1000);
		array[1] = new checkingAccount("1", "user", 1000);
		array[2] = new CheckingTrafficCardAccount("1", "user", 1000);
		array[3] = new CreditLineAccount("1", "user", 1000);
		array[4] = new BonusPointAccount("1", "user", 1000, 0);
		System.out.println();
		
		for (int i = 0; i < 5; i++) {
			if (array[i] instanceof Account) {
				array[i].deposit(1000);
				array[i].withdraw(1000);
				array[i].withdraw(5000);
				array[i].check();
				System.out.println();
			}
			
		}
		if (array[1] instanceof checkingAccount) {
			checkingAccount check = (checkingAccount) array[1]; // 다운캐스팅
			check.pay("1", 100);
			check.pay("2", 100);
		}
		System.out.println();
		if (array[2] instanceof CheckingTrafficCardAccount) {
			CheckingTrafficCardAccount traffic = (CheckingTrafficCardAccount) array[2];
			traffic.payTrafficCard("1", 100);
			traffic.payTrafficCard("2", 100);
		}
	}

}
